#### 登录获取requestId
```js
mounted(){
 if(this.timer){
     clearInterval(this.timer)
 }else{
     this.timer=setInterval(()=>{
        //  获取requestId的方法
         this.getRequestId()
     },10000)
 }
}
<!-- 回收 -->
destory(){
    clearInterval(this.timer)
}
```
